⚙️ كود برمجي (للمطورين)
═══════════════════════════════════════════════════════════════════════════════

إضافة مستخدم من خلال الكود:
──────────────────────────
from vip_subscription_system import SubscriptionManager

sm = SubscriptionManager()
success, msg = sm.add_user(
    user_id=123456789,
    username="ahmed_user",
    first_name="أحمد"
)

if success:
    print(msg)  # تم التسجيل بنجاح!
else:
    print(msg)  # خطأ


تفعيل أدمن من خلال الكود:
────────────────────────
from admin_panel import AdminPanel

admin = AdminPanel()
admin.add_admin(123456789)

if admin.is_admin(123456789):
    print("هو أدمن الآن!")


ترقية من خلال الكود:
──────────────────
success, msg = sm.upgrade_user(
    user_id=123456789,
    plan='gold',
    payment_method='admin'
)

if success:
    print(msg)  # تم الترقية!
